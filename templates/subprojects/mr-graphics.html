{{template "module.example_usage" .}}

// Initialize the graphics context
mr::graphics::Context context;
context.initialize(window.getNativeHandle());

// Create a renderer
mr::graphics::Renderer renderer(context);

// Load a mesh
auto mesh = mr::importer::load_mesh("models/character.gltf");

// Create material
auto material = renderer.createMaterial("shaders/pbr.glsl");
material->setTexture("albedo", loadTexture("textures/albedo.png"));
material->setTexture("normal", loadTexture("textures/normal.png"));
material->setTexture("metallic", loadTexture("textures/metallic.png"));
material->setTexture("roughness", loadTexture("textures/roughness.png"));

// Set up scene and camera
auto scene = renderer.createScene();
auto camera = scene->createCamera();
camera->setPosition({0, 1.7f, 5});
camera->lookAt({0, 0, 0});

// Add mesh to scene
auto entity = scene->createEntity();
entity->setMesh(mesh);
entity->setMaterial(material);

// Main render loop
while (window.isOpen()) {
    renderer.beginFrame();
    scene->update(deltaTime);
    renderer.renderScene(scene, camera);
    renderer.endFrame();
    window.swapBuffers();
}
